#include <ros.h>
#include <std_msgs/String.h>
 //Analog pins for current and voltage sensors
const int VIN_1 = A0; 
const int VIN_2 = A1;
const int VIN_3 = A2;
const int VIN_4 = A3;
const int VOLT_PIN = A4;
const float VCC   = 5.04;// supply voltage
const int MODEL = 17;   // enter the model (see above list)
        //using ACS770x-200B
        
int offset =20;// set the correction offset value
//Current and Voltage value variables
int wheel1_current;
int wheel2_current;
int wheel3_current;
int wheel4_current;
int volt_val;
double voltage;
#include <Robojax_AllegroACS_Current_Sensor.h>

Robojax_AllegroACS_Current_Sensor Wheel_1(MODEL,VIN_1);
Robojax_AllegroACS_Current_Sensor Wheel_2(MODEL,VIN_2);
Robojax_AllegroACS_Current_Sensor Wheel_3(MODEL,VIN_3);
Robojax_AllegroACS_Current_Sensor Wheel_4(MODEL,VIN_4);

void setup() {
    //Robojax.com Allegro ACS Current Sensor 
    Serial.begin(9600);// initialize serial monitor
    Serial.println("Robojax Tutorial");
    Serial.println("ACS770 Current Sensor");
  
}

void loop() {
   //Robojax.com Allegro ACS Current Sensor  
    wheel1_current = Wheel_1.getCurrent();
    wheel2_current = Wheel_2.getCurrent();
    wheel3_current = Wheel_3.getCurrent();
    wheel4_current = Wheel_4.getCurrent();
    volt_val = analogRead(VOLT_PIN);
    voltage = map(volt_val,0,1023,0,2500) + offset;
    Serial.print("Currents: ");
    Serial.print(wheel1_current); // print the current with 3 decimal places
    Serial.print(" ");
    Serial.print(wheel2_current); // print the current with 3 decimal places
    Serial.print(" ");
    Serial.print(wheel3_current); // print the current with 3 decimal places
    Serial.print(" ");
    Serial.print(wheel4_current); // print the current with 3 decimal places
    Serial.print(" ");
    Serial.println(voltage);
   
  delay(1500);
}
